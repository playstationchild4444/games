import tkinter as tk from tkinter import messagebox import random class Minesweeper: def __init__(self, master, rows=16, cols=16, mines=60): self.rows = rows self.cols = cols self.mines = mines self.master = master self.master.title("Minesweeper") self.grid = [[0 for _ in range(cols)] for _ in range(rows)] self.revealed = [[False for _ in range(cols)] for _ in range(rows)] self.game_over = False self.first_click = True # Create the grid of buttons self.buttons = [[None for _ in range(cols)] for _ in range(rows)] for i in range(rows): for j in range(cols): button = tk.Button(self.master, width=3, height=1, command=lambda row=i, col=j: self.click(row, col), relief="ridge", borderwidth=2) button.config(highlightthickness=0, bd=2, padx=5, pady=5) button.config(highlightbackground='#d9d9d9', highlightcolor='#d9d9d9') button.grid(row=i, column=j) self.buttons[i][j] = button # Plant mines randomly self.plant_mines() def plant_mines(self): mines_planted = 0 while mines_planted < self.mines: row = random.randint(0, self.rows - 1) col = random.randint(0, self.cols - 1) if self.grid[row][col] != -1: self.grid[row][col] = -1 mines_planted += 1 def click(self, row, col): if self.game_over: return if self.first_click: self.first_click = False # Reveal 3x3 area around first click for i in range(max(0, row - 1), min(self.rows, row + 2)): for j in range(max(0, col - 1), min(self.cols, col + 2)): if not self.revealed[i][j]: adjacent_mines = self.get_adjacent_mines(i, j) self.revealed[i][j] = True text = "" if adjacent_mines == 0 else str(adjacent_mines) bg_color = "green" if adjacent_mines == 0 else "lightgray" fg_color = "red" if adjacent_mines > 0 else "black" self.buttons[i][j].config(text=text, state=tk.DISABLED, bg=bg_color, fg=fg_color) if self.grid[row][col] == -1: self.reveal_all() self.game_over = True tk.messagebox.showinfo("Game Over", "You hit a mine!") return if self.revealed[row][col]: return self.revealed[row][col] = True adjacent_mines = self.get_adjacent_mines(row, col) text = "" if adjacent_mines == 0 else str(adjacent_mines) bg_color = "green" if adjacent_mines == 0 else "lightgray" fg_color = "red" if adjacent_mines > 0 else "black" self.buttons[row][col].config(text=text, state=tk.DISABLED, bg=bg_color, fg=fg_color) if self.get_adjacent_mines(row, col) == 0: self.reveal_adjacent(row, col) # Check if the game is won if all(self.revealed[i][j] for i in range(self.rows) for j in range(self.cols) if self.grid[i][j] != -1): self.game_over = True tk.messagebox.showinfo("You Win!", "Congratulations!") def get_adjacent_mines(self, row, col): mines = 0 for i in range(max(0, row - 1), min(self.rows, row + 2)): for j in range(max(0, col - 1), min(self.cols, col + 2)): if self.grid[i][j] == -1: mines += 1 return mines def reveal_adjacent(self, row, col): for i in range(max(0, row - 1), min(self.rows, row + 2)): for j in range(max(0, col - 1), min(self.cols, col + 2)): if not self.revealed[i][j] and self.grid[i][j] != -1: self.click(i, j) def reveal_all(self): for i in range(self.rows): for j in range(self.cols): if self.grid[i][j] == -1: self.buttons[i][j].config(text="*", state=tk.DISABLED, bg="red") else: adjacent_mines = self.get_adjacent_mines(i,j) text = "" if adjacent_mines == 0 else str(adjacent_mines) bg_color = "lightgray" fg_color = "red" if adjacent_mines > 0 else "black" self.buttons[i][j].config(text=text, state=tk.DISABLED, bg=bg_color, fg=fg_color) root = tk.Tk() game = Minesweeper(root) root.mainloop()
